php-fpm配置进程池优化：
pm               =dynamic; #dynamic，开启指定数量的php-fpm进程;dynamic,开始时开启一定数量的php-pfm进程，当请求量变大时，动态的增加php-pfm进程数到上限，空闲时自动释放空闲的进程到下限。

pm.max_children      =5; #静态方式下，开始启动5个进程。动态方式下限制最大启动进程数。
pm.start_servers     =2; #动态方式下起始php-fpm进程数量,改配置要求不能比min_spare_servers小，不能比max_spare_servers大
pm.min_spare_servers =1; #动态方式下空闲状态下的最小php-fpm进程数量
pm.max_spare_servers =3; #动态方式下空闲状态下的最大php-fpm进程数量
